002/**
003 * This class is part of the "World of Zuul" application. 
004 * "World of Zuul" is a very simple, text based adventure game.  
005 * 
006 * This class holds an enumeration table of all command words known to the game.
007 * It is used to recognise commands as they are typed in.
008 *
009 * @author  Michael Kolling and David J. Barnes + D.Bureau + B.VICTOIRE
010 * @version 2008.03.30 + 2019.09.25 + 2021.01.31 + 2021.01.09
011 */
012public class CommandWords
013{
014    // a constant array that will hold all valid command words
015    private static final String[] aValidCommands = { 
016        "go", "back", "look", "eat", "take", "drop", "help", "quit", "test"
017    };
018    
019
020    /**
021     * Check whether a given String is a valid command word. 
022     * @param pString a Command word to check
023     * @return true if a given string is a valid command,
024     * false if it isn't.
025     */
026    public boolean isCommand( final String pString )
027    {
028        for ( int i=0; i< CommandWords.aValidCommands.length; i++) {
029            if ( CommandWords.aValidCommands[i].equals(pString))
030                return true;
031        } // for
032        // if we get here, the string was not found in the commands
033        return false;
034    } // isCommand(.)
035    
036    /**
037     * @return all possible command word
038     */
039    public String getCommandList()
040    {
041        String vList = "";
042        for ( String vCom : aValidCommands ){
043            vList += vCom + " ";
044        }
045        return vList;
046    } // showAll()
047} // CommandWords
