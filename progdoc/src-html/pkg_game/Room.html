001package pkg_game;
002
003import java.util.HashMap;
004import pkg_game.pkg_item.ItemList;
005import pkg_game.pkg_item.Item;
006
007/**
008 * create and modify room
009 * 
010 * @author Brendan VICTOIRE
011 * @version 2021.01.18 + 2021.02.08
012 */
013public class Room {
014
015    private String aDescription;
016    private HashMap<String, Room> aExits;
017    private HashMap<String, Door> aDoors;
018    private String aImageName;
019    private ItemList aItems;
020
021    /**
022     * Constructor
023     * 
024     * @param pDescription description of the room
025     * @param pImage       image name
026     */
027    public Room(final String pDescription, final String pImage) {
028        this.aDescription = pDescription;
029        this.aExits = new HashMap<String, Room>();
030        this.aDoors = new HashMap<String, Door>();
031        this.aItems = new ItemList();
032        this.aImageName = pImage;
033    } // Room(.)
034
035    /**
036     * add exit to a room
037     * 
038     * @param pDrct     direction of the exit
039     * @param pNeighbor where the exit leads
040     */
041    public void setExit(final String pDrct, final Room pNeighbor) {
042        aExits.put(pDrct, pNeighbor);
043    } // setExits(.)
044
045    /**
046     * change the image
047     * 
048     * @param pImage name of the file
049     */
050    public void setImage(final String pImage) {
051        this.aImageName = pImage;
052    } // setImage(.)
053
054    /**
055     * add a door
056     * 
057     * @param pDrct the direction
058     * @param pDoor the door
059     */
060    public void setDoors(final String pDrct, final Door pDoor) {
061        this.aDoors.put(pDrct, pDoor);
062    }// setDoors
063
064    /**
065     * add item to a room
066     * 
067     * @param pItem item you want to add in the room
068     */
069    public void setItem(final Item pItem) {
070        this.aItems.setItem(pItem);
071    } // setItem(.)
072
073    /**
074     * access to description
075     * 
076     * @return a description
077     */
078    public String getDescription() {
079        return this.aDescription;
080    } // getDescription()
081
082    /**
083     * acces to the door
084     * 
085     * @param pDrct the direction
086     * @return the door
087     */
088    public Door getDoor(final String pDrct) {
089        return this.aDoors.get(pDrct);
090    }// getDoors
091
092    /**
093     * access map items of the rooms
094     * 
095     * @return map items of the rooms
096     */
097    public ItemList getItems() {
098        return this.aItems;
099    } // getItems()
100
101    /**
102     * access to the description + the exits
103     * 
104     * @return a string representation of the description
105     */
106    public String getLongDescription() {
107        return "You are " + this.aDescription + "\n" + this.getItemsString() + "\n" + this.getExitString();
108    } // getLongDescription()
109
110    /**
111     * @return true if the room has at least one exit
112     */
113    public boolean hasExit() {
114        return !aExits.isEmpty();
115    } // hasExit();
116
117    /**
118     * @return name of the image of the current room
119     */
120    public String getImageName() {
121        return this.aImageName;
122    } // getImageName()
123
124    /**
125     * acces to one exit
126     * 
127     * @param pDrct a direction
128     * @return the room where the exit leads
129     */
130    public Room getExit(final String pDrct) {
131        if (this.hasExit())
132            return this.aExits.get(pDrct);
133        else
134            return null;
135    } // getExit(.)
136
137    /**
138     * acces to all the exit under String format
139     * 
140     * @return exits under String format
141     */
142    public StringBuilder getExitString() {
143        StringBuilder vExits = new StringBuilder("");
144        if (this.hasExit()) {
145            vExits.append("Corridor:");
146            for (String vExit : this.aExits.keySet())
147                vExits.append(" " + vExit);
148        }
149        return vExits;
150    } // getExitString()
151
152    /**
153     * acces to the inventory under String format
154     * 
155     * @return inventory under String format
156     */
157    public StringBuilder getItemsString() {
158        StringBuilder vInventory = new StringBuilder("");
159        if (this.aItems.hasItem()) {
160            vInventory.append("Items:");
161            for (String vItem : this.aItems.getIventory().keySet())
162                vInventory.append(" " + vItem);
163        }
164        return vInventory;
165    } // getItemsString()
166
167    /**
168     * tells if the ItemList contains the item that we want
169     * 
170     * @param pItemName the name of the key that we want to check
171     * @return true if this room has this item
172     */
173    public boolean containsItem(final String pItemName) {
174        return this.aItems.containsItem(pItemName);
175    } // containsItem(.)
176
177    // Overriding equals() just in case it can be usefull later
178    @Override
179    public boolean equals(final Object pObj) {
180        if (pObj == this)
181            return true;
182        if (!(pObj.getClass().equals(this.getClass())))
183            return false;
184        Room vRoom = (Room) pObj;
185        return this.aDescription.equals(vRoom.getDescription()) && this.aExits.equals(vRoom.aExits);
186    } // equals(.)
187} // Room
