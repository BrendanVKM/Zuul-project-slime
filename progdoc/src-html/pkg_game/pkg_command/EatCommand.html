001package pkg_game.pkg_command;
002
003import pkg_game.GameEngine;
004
005public class EatCommand extends Command {
006    /**
007     * Execution of the Memorize command
008     */
009    @Override
010    public void execute(final Command pCom, final GameEngine pGameEngine) {
011        if (!pCom.hasSecondWord())
012            pGameEngine.getGui().println("What do you want to eat ?");
013        else {
014            String vSW = pCom.getSecondWord();
015            int vNumber = pGameEngine.getPlayer().eat(vSW);
016            if (vNumber == 0)
017                pGameEngine.getGui().println("You have eaten the Stone. Now you can carry more items.");
018            else if (vNumber == 1)
019                pGameEngine.getGui().println("You have eaten now and you are not hungry any more.");
020            else
021                pGameEngine.getGui().println("You don't carry this... if this exist.");
022        }
023    } // execute(.)
024} // EatCommand
