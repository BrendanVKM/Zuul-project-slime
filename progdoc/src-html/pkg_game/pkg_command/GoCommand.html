003import java.util.ArrayList;
004import java.util.Random;
005
006import pkg_game.GameEngine;
007
008/**
009 * Classe GoCom
010 *

012 */
013public class GoCommand extends Command {
014    /**
015     * Constructor
016     */
017    public GoCommand() {
018    } // GOCommand()
019
020    /**
021     * Execution of the Go command
022     */
023    @Override
024    public void execute(final Command pCom, final GameEngine pGameEngine) {
025        if (!pCom.hasSecondWord())
026            pGameEngine.getGui().println("Go where ?");
027        else {
028            if (pGameEngine.getPlayer().getCurrentRoom().equals(pGameEngine.getRooms().get("somewhere"))) {
029                ArrayList<String> vArray = new ArrayList<String>(pGameEngine.getRooms().keySet());
030                String vRR = pGameEngine.getNotRandomRoom();
031                if (vRR == null)
032                    vRR = vArray.get(new Random().nextInt(vArray.size()));
033                pGameEngine.getPlayer().goRandom(pGameEngine.getRooms().get(vRR));
034            } else {
035                String vDrct = pCom.getSecondWord();
036                int vNumber = pGameEngine.getPlayer().goRoom(vDrct);
037                if (vNumber == 0)
038                    pGameEngine.getGui().println("There is no corridor !");
039                else if (vNumber == 1) {
040                    pGameEngine.getGui().println("You can't pass without the dragonSoul");
041                }
042                if (pGameEngine.getVeldora() != null)
043                    if (pGameEngine.getPlayer().getCurrentRoom().equals(pGameEngine.getVeldora().getRoom()))
044                        pGameEngine.getGui().println(pGameEngine.getVeldora().VeldoraRoom());
045            }
046
047            if (pGameEngine.getPlayer().getCurrentRoom().equals(pGameEngine.getRooms().get("meetBeatrix"))) {
048                pGameEngine.getGui()
049                        .println("\n" + pGameEngine.getMovingCharacter("TabbySlime").PlayerMeetSlimeWithBeatrix());
050                pGameEngine.setNumberOfMeeting();
051            }
052
053            pGameEngine.slimeFollowing();
054        }
055    } // execute(.)
056}// GoCommand
