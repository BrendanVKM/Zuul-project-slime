001package pkg_game.pkg_command;
002
003import pkg_game.GameEngine;
004
005public class AleaCommand extends Command {
006    /**
007     * Execution of the Memorize command
008     */
009    @Override
010    public void execute(final Command pCom, final GameEngine pGameEngine) {
011        if (!pGameEngine.getTestMode())
012            pGameEngine.getGui().println("You need to be in the test mode!");
013        else if (!pCom.hasSecondWord()) {
014            pGameEngine.setNotRandomRoom(null);
015            pGameEngine.getGui().println("the randomness of the transporter room has been reset.");
016        } else
017            pGameEngine.setNotRandomRoom(pCom.getSecondWord());
018    } // execute(.)
019} // AleaCommand
