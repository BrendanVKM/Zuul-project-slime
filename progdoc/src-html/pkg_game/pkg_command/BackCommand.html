001package pkg_game.pkg_command;
002
003import pkg_game.GameEngine;
004
005public class BackCommand extends Command {
006    /**
007     * Execution of the Back command
008     */
009    @Override
010    public void execute(final Command pCom, final GameEngine pGameEngine) {
011        if (!pCom.hasSecondWord()) {
012            if (pGameEngine.getPlayer().getPreviousRoom() == null)
013                pGameEngine.getGui().println("You can't go back!");
014            else if (pGameEngine.getPlayer().getPreviousRoom().getImageName().equals(""))
015                pGameEngine.getGui().println("You were" + pGameEngine.getPlayer().getPreviousRoom().getDescription());
016            else if (pGameEngine.getPlayer().getCurrentRoom().equals(pGameEngine.getRooms().get("bottomOfLake"))
017                    && pGameEngine.getPlayer().getPreviousRoom().equals(pGameEngine.getRooms().get("cave7"))) {
018                pGameEngine.getPlayer().getPreviousRooms().clear();
019                pGameEngine.getGui().println("You were warned, you can't go back after jumping here.");
020            } else if (pGameEngine.getPlayer().getCurrentRoom().equals(pGameEngine.getRooms().get("cave5"))
021                    && pGameEngine.getPlayer().getPreviousRoom().equals(pGameEngine.getRooms().get("cave8"))) {
022                pGameEngine.getPlayer().getPreviousRooms().clear();
023                pGameEngine.getGui().println("You can't go back after falling here.");
024            } else {
025                pGameEngine.getPlayer().goBack();
026                pGameEngine.slimeFollowing();
027            }
028        } else
029            pGameEngine.getGui().println("Back what ?!");
030    } // execute(.)
031} // BackCommand
