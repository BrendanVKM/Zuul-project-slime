001package pkg_game.pkg_command;
002
003/**
004 * Representations for all the valid command words
005 * 
006 * @author Brendan VICTOIRE
007 * @version 2021.02.25
008 */
009public enum CommandWord {
010    UNKNOWN("?", null), GO("go", new GoCommand()), BACK("back", new BackCommand()),
011    MEMORIZE("memorize", new MemorizeCommand()), TELEPORT("teleport", new TeleportCommand()),
012    LOOK("look", new LookCommand()), ITEMS("items", new ItemsCommand()), TAKE("absorb", new TakeCommand()),
013    EAT("eat", new EatCommand()), DROP("drop", new DropCommand()), HELP("help", new HelpCommand()),
014    QUIT("quit", new QuitCommand()), TEST("test", new TestCommand()), ALEA("alea", new AleaCommand()),;
015
016    // The command string.
017    private String aCommandString;
018    private Command aCommand;
019
020    /**
021     * Initialise with the corresponding command word.
022     * 
023     * @param pCommandString The command string.
024     * @param pCommand the command 
025     */
026    CommandWord(final String pCommandString, final Command pCommand) {
027        this.aCommandString = pCommandString;
028        this.aCommand = pCommand;
029    }
030
031    /**
032     * access to the command
033     * 
034     * @return the command
035     */
036    public Command getCommand() {
037        return this.aCommand;
038    } // getCommand()
039
040    /**
041     * @return The command word as a string.
042     */
043    public String toString() {
044        return aCommandString;
045    } // toString
046}
