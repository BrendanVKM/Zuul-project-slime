001package pkg_game.pkg_command;
002
003import java.io.File;
004import java.io.FileNotFoundException;
005import java.util.Scanner;
006
007import pkg_game.GameEngine;
008
009public class TestCommand extends Command {
010    /**
011     * Execution of the Memorize command
012     */
013    @Override
014    public void execute(final Command pCom, final GameEngine pGameEngine) {
015        if (!pCom.hasSecondWord()) {
016            pGameEngine.getGui().println("Test What !?");
017            return;
018        }
019        Scanner vScanner;
020        try {
021            vScanner = new Scanner(new File("./test/" + pCom.getSecondWord() + ".txt"));
022            pGameEngine.setTestMode(true);
023            while (vScanner.hasNextLine()) {
024                String vLigne = vScanner.nextLine();
025                pGameEngine.interpretCommand(vLigne);
026            }
027            pGameEngine.setTestMode(false);
028        } catch (final FileNotFoundException pFNFE) {
029            pGameEngine.getGui().println("File not found.");
030        }
031    } // execute(.)
032} // TestCommand
