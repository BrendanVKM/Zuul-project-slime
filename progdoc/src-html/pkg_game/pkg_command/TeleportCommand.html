001package pkg_game.pkg_command;
002
003import pkg_game.GameEngine;
004
005public class TeleportCommand extends Command {
006    /**
007     * Execution of the Memorize command
008     */
009    @Override
010    public void execute(final Command pCom, final GameEngine pGameEngine) {
011        if (pGameEngine.getPlayer().getInventory().containsItem("teleportation")) {
012            if (!pCom.hasSecondWord()) {
013                if (pGameEngine.getPlayer().getBeamer().isUsed())
014                    pGameEngine.getGui().println("You already used it");
015                else {
016                    if (pGameEngine.getPlayer().teleport())
017                        pGameEngine.getPlayer().teleport();
018                    else
019                        pGameEngine.getGui().println("You need to memorize a room");
020                }
021            } else
022                pGameEngine.getGui().println("You can only teleport to your memorized room");
023        } else
024            pGameEngine.getGui().println("You cannot use that at the moment");
025    } // execute(.)
026} // TeleportCommand
