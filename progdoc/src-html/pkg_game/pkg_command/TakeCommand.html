001package pkg_game.pkg_command;
002
003import pkg_game.GameEngine;
004
005public class TakeCommand extends Command {
006    /**
007     * Execution of the Memorize command
008     */
009    @Override
010    public void execute(final Command pCom, final GameEngine pGameEngine) {
011        if (!pCom.hasSecondWord())
012            pGameEngine.getGui().println("Absorb what ?");
013        else {
014            String vSW = pCom.getSecondWord();
015            int vNumber = pGameEngine.getPlayer().take(vSW);
016            if (vNumber == 0)
017                pGameEngine.getGui().println("You don't have enough space to absorb this.");
018            else if (vNumber == 1) {
019                if (pGameEngine.getPlayer().getCurrentRoom().equals(pGameEngine.getRooms().get("onMagistone"))
020                        && vSW.equals("magistone")) {
021                    pGameEngine.getPlayer().getCurrentRoom().setImage("takeMagistone");
022                    pGameEngine.getGui().showImage(pGameEngine.getPlayer().getCurrentRoom().getImageName());
023                }
024                pGameEngine.getGui()
025                        .println("You absorb " + vSW + "\n" + pGameEngine.getPlayer().getCurrentRoom().getItemsString()
026                                + "\n" + pGameEngine.getPlayer().getInventoryString());
027            } else if (vNumber == 3) {
028                if (pGameEngine.getVeldora() != null) {
029                    pGameEngine.getPlayer().getInventory()
030                            .setItem(pGameEngine.getVeldora().getItems().getItem("dragonSoul"));
031                    pGameEngine.getGui().println(pGameEngine.getVeldora().PlayerAbsorbVeldora()
032                            + "\n You have acquired the sould of Veldora.");
033                    pGameEngine.nullVeldora();
034                }
035            } else
036                pGameEngine.getGui().println("What is this ? Is it there ?");
037        }
038    } // execute(.)
039} // TakeCommand
