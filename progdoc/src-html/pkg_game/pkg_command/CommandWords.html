001package pkg_game.pkg_command;
002
003import java.util.HashMap;
004
005/**
006 * This class is part of the "World of Zuul" application. "World of Zuul" is a
007 * very simple, text based adventure game.
008 * 
009 * This class holds an enumeration table of all command words known to the game.
010 * It is used to recognise commands as they are typed in.
011 *
012 * @author Michael Kolling and David J. Barnes + D.Bureau + B.VICTOIRE
013 * @version 2008.03.30 + 2019.09.25 + 2021.01.31 + 2021.01.09 + 2021.02.24 +
014 *          2021.02.26
015 */
016public class CommandWords {
017    private HashMap<String, CommandWord> aValidCommands;
018
019    /**
020     * Constructor
021     */
022    public CommandWords() {
023        this.aValidCommands = new HashMap<String, CommandWord>();
024        for (CommandWord command : CommandWord.values())
025            if (command != CommandWord.UNKNOWN)
026                this.aValidCommands.put(command.toString(), command);
027    } // CommandWords()
028
029    /**
030     * Find the CommandWord associated with a command word.
031     * 
032     * @param pCommandWord The word to look up.
033     * @return The CommandWord correspondng to commandWord, or UNKNOWN if it is not
034     *         a valid command word.
035     */
036    public Command getCommandWord(final String pCommandWord) {
037        if (this.aValidCommands.get(pCommandWord) != null)
038            return (Command) this.aValidCommands.get(pCommandWord).getCommand();
039        else
040            return null;
041    } // getCommandWord(.)
042
043    /**
044     * Check whether a given String is a valid command word.
045     * 
046     * @param pString a Command word to check
047     * @return true if a given string is a valid command, false if it isn't.
048     */
049    public boolean isCommand(final String pString) {
050        return this.aValidCommands.containsKey(pString);
051    } // isCommand(.)
052
053    /**
054     * acces to all the valid commands words
055     * 
056     * @return all possible command words
057     */
058    public String getCommandList() {
059        String vList = "";
060        for (String vCom : this.aValidCommands.keySet()) {
061            vList += vCom + " ";
062        }
063        return vList;
064    } // showAll()
065} // CommandWords
