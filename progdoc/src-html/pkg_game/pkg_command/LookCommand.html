001package pkg_game.pkg_command;
002
003import pkg_game.GameEngine;
004
005public class LookCommand extends Command {
006    /**
007     * Execution of the Memorize command
008     */
009    @Override
010    public void execute(final Command pCom, final GameEngine pGameEngine) {
011        if (pCom.hasSecondWord()) {
012            String vSW = pCom.getSecondWord();
013            pGameEngine.getGui().print("You are looking at ");
014            if (pGameEngine.getPlayer().look(vSW))
015                pGameEngine.getGui().print("a " + pGameEngine.getPlayer().getCurrentRoom().getItems().getItem(vSW).getItemString() + "\n");
016            else
017                pGameEngine.getGui().println("... I don't know what you are looking at! Is this in here?");
018        } else {
019            if (pGameEngine.getPlayer().getCurrentRoom().getImageName().equals("cave6"))
020                pGameEngine.getPlayer().changeImage("lookAtTheCave");
021            pGameEngine.getGui().println(pGameEngine.getPlayer().getCurrentRoom().getLongDescription());
022        }
023    } // execute(.)
024} // LookCommand
