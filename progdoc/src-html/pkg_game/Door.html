001package pkg_game;
002
003import pkg_game.pkg_item.Item;
004
005/**
006 * set Door to room
007 * 
008 * @author Brendan VICTOIRE
009 * @version 2021.02.26
010 */
011public class Door {
012    private boolean aOpen;
013    private Item aKey;
014
015    /**
016     * defalt constructor
017     * 
018     * @param pKey the key that you need to open the door
019     */
020    public Door(final Item pKey) {
021        this.aKey = pKey;
022        this.aOpen = false;
023    } // Door(.)
024
025    /**
026     * acces to the key
027     * 
028     * @return the key
029     */
030    public Item getKey() {
031        return aKey;
032    } // getKey()
033
034    /**
035     * acces to the status of the door (close or open)
036     * 
037     * @return true if the door is open false otherwise
038     */
039    public boolean isOpen() {
040        return this.aOpen;
041    } // isOpen()
042
043    /**
044     * acces to the status of the door (close or open)
045     * 
046     * @param pOpen true if the door is open false otherwise
047     */
048    public void setOpen(final boolean pOpen) {
049        this.aOpen = pOpen;
050    } // setOpen(.)
051} // Door
