002/**
003 * let you teleport in a room that you have memorized
004 * @author Brendan VICTOIRE
005 * @version 2021.02.26
006 */
007public class Beamer extends Item
008{
009    private Room aMemorizedRoom;
010    private boolean aCharged;
011    private boolean aUsed;
012
013    public Beamer()
014    {
015        super( "teleportation", "let you teleport in a room that you have memorized (one-time use)", 0 );
016        this.aMemorizedRoom = null;
017        this.aCharged = false;
018        this.aUsed = false;
019    } // Beamer()
020
021    /**
022     * access to the memorized room 
023     * @return the memorized room
024     */
025    public Room getMemorizedRoom()
026    {
027        return this.aMemorizedRoom;
028    }
029    /**
030     * access to aCharged 
031     * @return true if the beamer if charged
032     */
033    public boolean isCharged()
034    {
035        return this.aCharged;
036    } // isCharged()
037
038    /**
039     * know if the beamer has already been used or not
040     * @return true if the beamer has been used
041     */
042    public boolean isUsed()
043    {
044        return this.aUsed;
045    } // isUsed()
046
047    /**
048     * charge the beamer with a room
049     * @param pRoom the room that you want to memorized
050     */
051    public void charge( final Room pRoom )
052    {
053        this.aMemorizedRoom = pRoom;
054        if ( this.aMemorizedRoom != null )
055            this.aCharged = true;
056    } // charge(.)
057
058    public void fire()
059    {
060        this.aMemorizedRoom = null;
061        this.aCharged = false;
062        this.aUsed = true;
063    }
064} // Beamer
