166    /**
167     * interpret commands
168     * stop the game if you have executed to much command
169     * @param pCom a command
170     */
171    public void interpretCommand( final String pCom ) 
172    {
173        this.aGui.println( "\n" + "> " + pCom );
174        Command vCom = this.aParser.getCommand( pCom );
175        CommandWord vCW = vCom.getCommandWord();
176
177        if ( this.aPlayer.commandCounter() ){
178            this.aGui.println( "You are too tired to continue.");
179            this.aPlayer.quit( new Command( CommandWord.QUIT, null ) );
180            return;
181        }
182
183        switch ( vCW ){
184            case UNKNOWN : 
185                this.aGui.println("I don't know what you mean...");
186                break;
187            case GO:
188                this.aPlayer.goRoom(vCom);
189                break;
190            case BACK:
191                this.aPlayer.goBack(vCom);
192                break;
193            case LOOK:
194                this.aPlayer.look(vCom);
195                break;
196            case ITEMS:
197                this.aPlayer.Items(vCom);
198                break;
199            case TAKE:
200                this.aPlayer.take(vCom);
201                break;
202            case EAT:
203                this.aPlayer.eat(vCom);
204                break;
205            case DROP:
206                this.aPlayer.drop(vCom);
207                break;
208            case HELP:
209                this.aPlayer.printHelp();
210                break;
211            case QUIT:
212                this.aPlayer.quit(vCom);
213                break;
214            case TEST:
215                this.test(vCom);
216                break;
217        }
218    } // interpretCommand(.)

231    /**
232     * Execute test 
233     * @param pCom a command
234     */
235    private void test ( final Command pCom )
236    {
237        if ( !pCom.hasSecondWord() ){
238            this.aGui.println( "Test What !?" );
239            return;
240        }
241        Scanner vScanner;
242        try{
243            vScanner = new Scanner( new File( "./test/" + pCom.getSecondWord() + ".txt" ) );
244            while ( vScanner.hasNextLine() ){
245                String vLigne = vScanner.nextLine();
246                this.interpretCommand( vLigne );
247            }
248        }
249        catch ( final FileNotFoundException pFNFE ){
250            this.aGui.println( "File not found." );
251        }
252    } // test(.)
