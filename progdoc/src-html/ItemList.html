001import java.util.HashMap ;
002
003/**
004 * create and modify Itemlist for Player and Room
005 * @author Brendan VICTOIRE
006 * @version 2021.01.18 + 2021.02.08
007 */
008public class ItemList {
009    private HashMap<String,Item> aInventory ;
010
011    /**
012     * default constructor
013     */
014    public ItemList()
015    {
016        this.aInventory = new HashMap<String, Item>();
017    } // itemList
018
019    /**
020     * add an item to the room
021     * @param pItem an item
022     */
023    public void setItem( final Item pItem ) 
024    {
025        this.aInventory.put(pItem.getName(), pItem);
026    } // setItem(.)
027
028    /**
029     * remove an item
030     * @param pItem the name of the item
031     */
032    public void removeItem( final String pItem ) 
033    {
034        this.aInventory.remove(pItem);
035    } // removeItem(.)
036
037    /**
038     * acces to all the inventory
039     * @return inventory map
040     */
041    public HashMap<String, Item> getIventory(){ return this.aInventory; } // getIventory()
042
043    /**
044     * tells if the ItemList is empty or not
045     * @return true if the room has at least one item
046     */
047    public boolean hasItem() { return !aInventory.isEmpty() ; } // hasItem()
048    
049    /**
050     * tells if the ItemList contains the item that we want
051     * @param pItemName the name of the key that we want to check
052     * @return true if this room has this item
053     */
054    public boolean containsItem( final String pItemName )
055    {
056        return this.aInventory.containsKey( pItemName );
057    } // containsItem(.)
058
059    /**
060     * acces to the item that we want
061     * @param pItem an item
062     * @return the item
063     */
064    public Item getItem( final String pItem )
065    {
066        if ( this.hasItem() ) {
067            return this.aInventory.get(pItem);
068        }
069        else return null; 
070    } // getItem(.)
071
072    /**
073     * acces to the total weight of the ItemList
074     * @return the weight the player is carrying
075     */
076    public double getWeightCarrying()
077    {
078        if ( this.hasItem() ){
079            double vWeight = 0;
080            for ( Item vItem : this.aInventory.values())
081                vWeight += vItem.getWeight();
082            return vWeight;
083        }
084        else return 0;
085    } // getWeightCarrying()
086}
