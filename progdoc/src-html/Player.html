001import java.util.Stack;
002
003public class Player {
004    private Room aCurrentRoom;
005    private Stack<Room> aPreviousRooms;
006    private ItemList aInventory;
007    private UserInterface aGui;
008    private Parser aParser;
009
010    /**
011     * Create the player
012     */
013    public Player()
014    {
015        this.aPreviousRooms = new Stack<Room>() ;
016        this.aInventory = new ItemList();
017        this.aParser = new Parser();
018    } // Player()
019
020    /**
021     * set the GUI
022     * @param pUserInterface the user interface
023     */
024    public void setGUI( final UserInterface pUserInterface )
025    {
026        this.aGui = pUserInterface;
027    } // setGUI(.)
028
029    /**
030     * @return Current room
031     */
032    public Room getCurrentRoom(){ return this.aCurrentRoom; } // getCurrentRoom()
033
034    /**
035     * @return Previous room
036     */
037    public Room getPreviousRoom(){ return this.aPreviousRooms.peek(); } // getPreviousRoom()
038
039    /**
040     * @return stack of previous rooms
041     */
042    public Stack<Room> getPreviousRooms(){ return this.aPreviousRooms; } // getPreviousRoom()
043    /**
044     * Set the current room
045     * @param pCurrentRoom the current room
046     */
047    public void setCurrentRoom(final Room pCurrentRoom)
048    {
049        this.aCurrentRoom = pCurrentRoom ;
050    } // setCurrentRoom(.)
051
052    /**
053     * @param vNextRoom next room
054     */
055    public void nextRoom( final Room pNextRoom )
056    {
057        this.aPreviousRooms.push( this.aCurrentRoom ); 
058        this.aCurrentRoom = pNextRoom;
059    } // nextRoom(.)
060
061    /**
062     * change the list of previous rooms
063     */
064    public void previousRooms(){
065        nextRoom( this.aPreviousRooms.peek() );
066        this.aPreviousRooms.pop();
067    } // previousRooms(.)
068    
069    /**
070     * print location info
071     */
072    void printLocationInfo()
073    {
074        this.aGui.println( "You were " + this.getPreviousRoom().getDescription() );
075        this.aGui.println( this.aCurrentRoom.getLongDescription() );
076        if ( this.aInventory.hasItem() )
077            this.aGui.println( "" + this.aInventory.getItemsString() );
078        if ( this.aCurrentRoom.getImageName() != null )
079            this.aGui.showImage( this.aCurrentRoom.getImageName() );
080    } // printLocationInfo()
081
082    /**
083     * method exected with the command word "go"
084     * @param pDep command to move to the next
085     */
086    public void goRoom( final Command pDep )
087    {
088        if ( !pDep.hasSecondWord() )
089            this.aGui.println( "Go where ?" );
090        else {
091            String vDrct = pDep.getSecondWord().toLowerCase(); 
092            Room vNextRoom = this.aCurrentRoom.getExit(vDrct);
093            if ( vNextRoom == null ) this.aGui.println( "There is no door !" );
094            else {
095                this.nextRoom( vNextRoom );
096                this.printLocationInfo();
097            }
098        }
099    } // goRoom(.)
100
101    /**
102     * go backward
103     * @param pCom a command
104     */
105    public void goBack( final Command pCom )
106    {
107        if (this.aCurrentRoom.getDescription().equals( "somewhere in a cave asking yourself where is Beatrix." )
108            && this.aPreviousRooms.size() == 1)
109                this.aPreviousRooms.clear();
110
111        if ( !pCom.hasSecondWord() ){
112            if ( this.aPreviousRooms.empty() ) 
113                this.aGui.println( "You can't go back!" );
114            //if you are in the bottom of the lake you can't go back to the cave you have to go further
115            else if ( this.aCurrentRoom.getDescription().equals( "deep down in a lake." ) 
116                && this.getPreviousRoom().getDescription().equals( "somewhere in a cave. Be careful if you jump, you won't be able to come back here." ) )
117                    this.aPreviousRooms.clear();
118            else if ( this.aPreviousRooms.empty() ) 
119                this.aGui.println( "That's where you wake up you can't go back anymore.");
120            else {
121                this.previousRooms();
122                this.printLocationInfo();
123            }
124        }
125        else this.aGui.println( "Back what ?!" );
126    } // goBack(.)
127
128    /**
129     * look 
130     * @param pCom a command
131     */
132    public void look( final Command pCom )
133    {
134        if ( pCom.hasSecondWord() ){
135            String vSW = pCom.getSecondWord();
136            this.aGui.print("You are looking at ");
137            if ( this.aCurrentRoom.getItems().containsItem(vSW) ){
138                this.aGui.print( "a " + this.aCurrentRoom.getItems().getItem(vSW).getItemString() + "\n");
139                return;
140            }
141            this.aGui.println("... I don't know what you are looking at !");
142        }
143        else this.aGui.println( this.aCurrentRoom.getLongDescription() );
144    } // look(.)
145
146    /**
147     * eat
148     * @param pCom a command
149     */
150    public void eat( final Command pCom )
151    {
152        if ( !pCom.hasSecondWord() )
153            this.aGui.println( "What do you want to eat ?");
154        else this.aGui.println( "You have eaten now and you are not hungry any more.");
155    } // eat(.)
156
157    /**
158     * take an item
159     * @param pCom a command
160     */
161    public void take( final Command pCom )
162    {
163        if ( !pCom.hasSecondWord() )
164            this.aGui.println( "Take what ?");
165        else {
166            String vSW = pCom.getSecondWord();
167            if ( this.aCurrentRoom.getItems().containsItem(vSW) ){
168                this.aInventory.setItem( this.getCurrentRoom().getItems().getItem(vSW) );
169                this.aCurrentRoom.getItems().removeItem(vSW);
170                this.aGui.println( "You take " + vSW + "\n"
171                    + this.aCurrentRoom.getItems().getItemsString() + "\n"
172                    +this.aInventory.getItemsString() );
173            }
174            else this.aGui.println( "What is this ? Is it there ?" );
175        }
176    } // take(.)
177
178    /**
179     * drop an item
180     * @param pItem the name of the item
181     */
182    public void drop( final Command pCom )
183    {
184        if ( !pCom.hasSecondWord() )
185            this.aGui.println( "Drop what ?");
186        else {
187            String vSW = pCom.getSecondWord();
188            if ( this.aInventory.containsItem(vSW) ){
189                this.aCurrentRoom.getItems().setItem( this.aInventory.getItem(vSW) );
190                this.aInventory.removeItem(vSW);
191                this.aGui.println( "You droped " + vSW + "\n"
192                    + this.aCurrentRoom.getItems().getItemsString() + "\n"
193                    + this.aInventory.getItemsString() );
194            }
195            else this.aGui.println( "What is this ? Do you even have this ?" );
196        }
197    } // drop(.)
198    
199    /**
200     * help message
201     */
202    public void printHelp()
203    {
204        this.aGui.println( "You are lost. You are alone." );
205        this.aGui.println( "You wander around at the cave.\n" );
206        this.aGui.println( "Your command words are:" );
207        this.aGui.println( this.aParser.getCommandString() );
208    } // printHelp()
209
210    /**
211     * quit the game
212     * @param pCom a command 
213     * @return true if the Command doesn't have a second word
214     */
215    public void quit( final Command pCom )
216    {
217        if ( pCom.hasSecondWord() )
218                this.aGui.println( "Quit what?" );
219        else{
220            this.aGui.println( "Thank you for playing.  Good bye." );
221            this.aGui.enable( false );
222        }
223    } // quit(.)
224} // Player
