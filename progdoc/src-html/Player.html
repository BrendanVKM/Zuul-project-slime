001import java.util.Stack;
002
003/**
004 * create and modify the player and his inventory
005 * @author Brendan VICTOIRE
006 * @version 2021.02.23 + 2021.02.25
007 */
008public class Player {
009    private Room aCurrentRoom;
010    private Stack<Room> aPreviousRooms;
011    private ItemList aInventory;
012    private UserInterface aGui;
013    private Parser aParser;
014    private int aCommandEntered;
015    private double MAX_WEIGHT = 1000; // a slime can carry anything
016
017    /**
018     * Create the player
019     * default constructor
020     */
021    public Player()
022    {
023        this.aPreviousRooms = new Stack<Room>() ;
024        this.aInventory = new ItemList();
025        this.aParser = new Parser();
026        this.aCommandEntered = 0;
027    } // Player()
028
029    /**
030     * set the GUI
031     * @param pUserInterface the user interface
032     */
033    public void setGUI( final UserInterface pUserInterface )
034    {
035        this.aGui = pUserInterface;
036    } // setGUI(.)
037
038    /**
039     * access the current room
040     * @return Current room
041     */
042    public Room getCurrentRoom(){ return this.aCurrentRoom; } // getCurrentRoom()
043
044    /**
045     * 
046     * access the previous room
047     * @return Previous room
048     */
049    public Room getPreviousRoom(){ return this.aPreviousRooms.peek(); } // getPreviousRoom()
050
051    /**
052     * access stack of previous rooms 
053     * @return stack of previous rooms
054     */
055    public Stack<Room> getPreviousRooms(){ return this.aPreviousRooms; } // getPreviousRoom()
056    
057    /**
058     * print the content of the invetory
059     * @return inventory under String format
060     */
061    public StringBuilder getInventoryString()
062    {
063        StringBuilder vInventory = new StringBuilder( "" );
064        if (this.aInventory.hasItem() ){
065            vInventory.append( "Iventory:" );
066            for ( String vItem : this.aInventory.getIventory().keySet() )
067            vInventory.append( " " + vItem );
068        }
069        return vInventory;
070    } // getInventoryString()
071
072    /**
073     * Set the current room
074     * @param pCurrentRoom the current room
075     */
076    public void setCurrentRoom(final Room pCurrentRoom)
077    {
078        this.aCurrentRoom = pCurrentRoom ;
079    } // setCurrentRoom(.)
080
081    /**
082     * set the next room as the current room
083     * @param pNextRoom next room
084     */
085    public void setNextRoom( final Room pNextRoom )
086    {
087        this.aPreviousRooms.push( this.aCurrentRoom ); 
088        this.aCurrentRoom = pNextRoom;
089    } // setNextRoom(.)
090
091    /**
092     * change the list of previous rooms
093     */
094    public void setPreviousRooms(){
095        this.setNextRoom( this.aPreviousRooms.peek() );
096        this.aPreviousRooms.pop();
097        this.aPreviousRooms.pop();
098    } // setPreviousRooms(.)
099    
100    /**
101     * print location info (current room description + inventory of this player)
102     */
103    void printLocationInfo()
104    {
105        this.aGui.println( "You were " + this.getPreviousRoom().getDescription() );
106        this.aGui.println( this.aCurrentRoom.getLongDescription() );
107        if ( this.aInventory.hasItem() )
108            this.aGui.println( "" + this.aCurrentRoom.getItemsString() );
109        if ( this.aCurrentRoom.getImageName() != null )
110            this.aGui.showImage( this.aCurrentRoom.getImageName() );
111    } // printLocationInfo()
112
113    /**
114     * limit the number of command that a player can type before force ending
115     * @return true if the limit is reached
116     */
117    public boolean commandCounter()
118    {
119        this.aCommandEntered++;
120        return this.aCommandEntered > 30;
121    }
122    /**
123     * method executed with the command word "go"
124     * let you go forward
125     * @param pDep command to move to the next
126     */
127    public void goRoom( final Command pDep )
128    {
129        if ( !pDep.hasSecondWord() )
130            this.aGui.println( "Go where ?" );
131        else {
132            String vDrct = pDep.getSecondWord(); 
133            Room vNextRoom = this.aCurrentRoom.getExit(vDrct);
134            if ( vNextRoom == null ) this.aGui.println( "There is no door !" );
135            else {
136                this.setNextRoom( vNextRoom );
137                this.printLocationInfo();
138            }
139        }
140    } // goRoom(.)
141
142    /**
143     * method executed with the command word "back"
144     * let you go backward
145     * @param pCom a command
146     */
147    public void goBack( final Command pCom )
148    {
149        if ( !pCom.hasSecondWord() ){
150            if ( this.getPreviousRoom().getDescription().equals( "... Wait! Where are you?" )) 
151                this.aGui.println( "You were" + this.getPreviousRoom().getDescription() );
152            //if you are in the bottom of the lake you can't go back to the cave you have to go further
153            else if ( this.aCurrentRoom.getDescription().equals( "deep down in a lake." ) 
154                && this.getPreviousRoom().getDescription().equals( "somewhere in a cave. Be careful if you jump, you won't be able to come back here." ) ){
155                    this.aPreviousRooms.clear();
156                    this.aGui.println( "You were warned, you can't go back after jumping here." );
157                }
158            else if ( this.aCurrentRoom.getDescription().equals( "somewhere in a cave. You see a corridor that you can't reach" ) 
159                && this.getPreviousRoom().getDescription().equals( "somewhere in a cave. You have crossed the secret path." ) ){
160                    this.aPreviousRooms.clear();
161                    this.aGui.println( "You can't go back after falling here." );
162                }
163            else {
164                this.setPreviousRooms();
165                this.printLocationInfo();
166            }
167        }
168        else this.aGui.println( "Back what ?!" );
169    } // goBack(.)
170
171    /**
172     * method executed with the command word "look"
173     * print all the information of the room or the item that you are looking at
174     * @param pCom a command
175     */
176    public void look( final Command pCom )
177    {
178        if ( pCom.hasSecondWord() ){
179            String vSW = pCom.getSecondWord();
180            this.aGui.print("You are looking at ");
181            if ( this.aCurrentRoom.getItems().containsItem(vSW) ){
182                this.aGui.print( "a " + this.aCurrentRoom.getItems().getItem(vSW).getItemString() + "\n");
183                return;
184            }
185            this.aGui.println("... I don't know what you are looking at !");
186        }
187        else this.aGui.println( this.aCurrentRoom.getLongDescription() );
188    } // look(.)
189
190    /**
191     * method executed with the command word "eat"
192     * if possible eat an item from your inventory
193     * @param pCom a command
194     */
195    public void eat( final Command pCom )
196    {
197        if ( !pCom.hasSecondWord() )
198            this.aGui.println( "What do you want to eat ?");
199        else {
200            String vSW = pCom.getSecondWord();
201            if ( this.aInventory.containsItem(vSW) ){
202                if ( vSW.equals("dragonSoul") ){
203                    this.aGui.println( "You have eaten the Veldora's soul. Now you can carry more items." );
204                    this.MAX_WEIGHT *= 1e7;
205                }
206                else this.aGui.println( "You have eaten now and you are not hungry any more.");
207            this.aInventory.removeItem(vSW);
208            }
209            else this.aGui.println( "You don't carry this ");
210        }
211    } // eat(.)
212
213    /**
214     * method executed with the command word "items" 
215     * print a list of the item carried by the player
216     * @param pCom a command
217     */
218    public void Items( final Command pCom )
219    {
220        if ( pCom.hasSecondWord())
221            this.aGui.println( "Items what?" );
222        else if ( this.aInventory.hasItem()){
223            StringBuilder vInventory = new StringBuilder( " " );
224            if (this.aInventory.hasItem() ){
225                vInventory.append( "Iventory:" );
226                for ( String vItem : this.aInventory.getIventory().keySet() )
227                    vInventory.append( " " + vItem + " (" + this.aInventory.getItem(vItem).getWeight() + ")" );
228                this.aGui.println( "" + vInventory + " for a total of "+ this.aInventory.getWeightCarrying() + "kg" );
229            }
230            else this.aGui.println( "" + vInventory );
231        }
232        else this.aGui.println( "You aren't carrying anything." );
233    } // Items()
234
235    /**
236     * method executed with the command word "take" 
237     * if possible take an item and put it in the inventory of the player
238     * @param pCom a command
239     */
240    public void take( final Command pCom )
241    {
242        if ( !pCom.hasSecondWord() )
243            this.aGui.println( "Take what ?");
244        else {
245            String vSW = pCom.getSecondWord();
246            if ( this.aCurrentRoom.getItems().containsItem(vSW) ){
247                if ( this.aInventory.getWeightCarrying() < this.MAX_WEIGHT)
248                    if ( ( this.aInventory.getWeightCarrying() + this.aCurrentRoom.getItems().getItem(vSW).getWeight() ) > this.MAX_WEIGHT )
249                        this.aGui.println( "You don't have enough space to take this");
250                    else {
251                        this.aInventory.setItem( this.getCurrentRoom().getItems().getItem(vSW) );
252                        this.aCurrentRoom.getItems().removeItem(vSW);
253                        this.aGui.println( "You take " + vSW + "\n"
254                            + this.aCurrentRoom.getItemsString() + "\n"
255                            +this.getInventoryString() );
256                    }
257            }
258            else this.aGui.println( "What is this ? Is it there ?" );
259        }
260    } // take(.)
261
262    /**
263     * method executed with the command word "drop" 
264     * if possible drop an item on the floor of the room
265     * @param pItem the name of the item
266     */
267    public void drop( final Command pCom )
268    {
269        if ( !pCom.hasSecondWord() )
270            this.aGui.println( "Drop what ?");
271        else {
272            String vSW = pCom.getSecondWord();
273            if ( this.aInventory.containsItem(vSW) ){
274                this.aCurrentRoom.getItems().setItem( this.aInventory.getItem(vSW) );
275                this.aInventory.removeItem(vSW);
276                this.aGui.println( "You droped " + vSW + "\n"
277                    + this.aCurrentRoom.getItemsString() + "\n"
278                    + this.getInventoryString() );
279            }
280            else this.aGui.println( "What is this ? Do you even have this ?" );
281        }
282    } // drop(.)
283

285     * method executed with the command word "help" 
286     * print help message
287     */
288    public void printHelp()
289    {
290        this.aGui.println( "You are lost. You are alone." );

292        this.aGui.println( "Your command words are:" );
293        this.aGui.println( this.aParser.getCommandString() );
294    } // printHelp()

296    /**
297     * method executed with the command word "quit" 
298     * if right executed quit the game
299     * @param pCom a command 
300     */
301    public void quit( final Command pCom )
302    {
303        if ( pCom.hasSecondWord() )
304                this.aGui.println( "Quit what?" );
305        else{
306            this.aGui.println( "Thank you for playing.  Good bye." );
307            this.aGui.enable( false );
308        }
309    } // quit(.)
310} // Player
