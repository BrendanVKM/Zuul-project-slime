001import java.util.Stack;
002
003/**
004 * create and modify the player and his inventory
005 * @author Brendan VICTOIRE
006 * @version 2021.02.23 + 2021.02.25 +2021.02.26
007 */
008public class Player {
009    private Room aCurrentRoom;
010    private Stack<Room> aPreviousRooms;
011    private ItemList aInventory;
012    private UserInterface aGui;
013    private Beamer aBeamer;
014    private int ENTERED_COMMAND = 2500;
015    private double MAX_WEIGHT = 1e10; // a slime can carry anything
016
017    /**
018     * Create the player
019     * default constructor
020     */
021    public Player()
022    {
023        this.aPreviousRooms = new Stack<Room>() ;
024        this.aInventory = new ItemList();
025        this.aBeamer = new Beamer();
026    } // Player()
027
028    /**
029     * set the GUI
030     * @param pUserInterface the user interface
031     */
032    public void setGUI( final UserInterface pUserInterface )
033    {
034        this.aGui = pUserInterface;
035    } // setGUI(.)
036
037    /**
038     * access the current room
039     * @return Current room
040     */
041    public Room getCurrentRoom(){ return this.aCurrentRoom; } // getCurrentRoom()
042
043    /**
044     * 
045     * access the previous room
046     * @return Previous room
047     */
048    public Room getPreviousRoom(){ return this.aPreviousRooms.peek(); } // getPreviousRoom()
049
050    /**
051     * access stack of previous rooms 
052     * @return stack of previous rooms
053     */
054    public Stack<Room> getPreviousRooms(){ return this.aPreviousRooms; } // getPreviousRoom()
055    
056    /**
057     * access to the player inventory
058     * @return the player inventory
059     */
060    public ItemList getInventory(){ return this.aInventory; } // getInventory()
061
062    /**
063     * acces to the beamer
064     * @return the beamer
065     */
066    public Beamer getBeamer(){ return this.aBeamer; } // getBeamer()
067
068    /**
069     * acces to the total weight of the ItemList
070     * @return the weight the player is carrying
071     */
072    public double getWeightCarrying()
073    {
074        if ( this.aInventory.hasItem() ){
075            double vWeight = 0;
076            for ( Item vItem : this.aInventory.getIventory().values())
077                vWeight += vItem.getWeight();
078            return vWeight;
079        }
080        else return 0;
081    } // getWeightCarrying()
082
083    /**
084     * print the content of the invetory
085     * @return inventory under String format
086     */
087    public StringBuilder getInventoryString()
088    {
089        StringBuilder vInventory = new StringBuilder( " " );
090                if (this.aInventory.hasItem() ){
091                    vInventory.append( "Iventory:" );
092                for ( String vItem : this.aInventory.getIventory().keySet() )
093                    vInventory.append( " " + vItem + " (" + this.aInventory.getItem(vItem).getWeight() + ")" );
094                return vInventory.append( "\nfor a total of "+ this.getWeightCarrying() + "kg" );
095            }
096            return vInventory;
097    } // getInventoryString()
098
099    /**
100     * Set the current room
101     * @param pCurrentRoom the current room
102     */
103    public void setCurrentRoom(final Room pCurrentRoom)
104    {
105        this.aCurrentRoom = pCurrentRoom ;
106    } // setCurrentRoom(.)
107
108    /**
109     * set the next room as the current room
110     * @param pNextRoom next room
111     */
112    public void setNextRoom( final Room pNextRoom )
113    {
114        this.aPreviousRooms.push( this.aCurrentRoom ); 
115        this.aCurrentRoom = pNextRoom;
116    } // setNextRoom(.)
117
118    /**
119     * change the list of previous rooms
120     */
121    public void setPreviousRooms(){
122        this.setNextRoom( this.aPreviousRooms.peek() );
123        this.aPreviousRooms.pop();
124        this.aPreviousRooms.pop();
125    } // setPreviousRooms(.)
126    
127    /**
128     * print location info (current room description + inventory of this player)
129     */
130    public void printLocationInfo()
131    {
132        this.aGui.println( "You were " + this.getPreviousRoom().getDescription() );
133        this.aGui.println( this.aCurrentRoom.getLongDescription() );
134        if ( this.aInventory.hasItem() )
135            this.aGui.println( "" + this.aCurrentRoom.getItemsString() );
136        if ( this.aCurrentRoom.getImageName() != "" )
137            this.aGui.showImage( this.aCurrentRoom.getImageName() );
138    } // printLocationInfo()
139    
140    /**
141     * limit the number of command that a player can type before force ending
142     * @return true if the limit is reached
143     */
144    public boolean commandCounter()
145    {
146        this.ENTERED_COMMAND--;
147        return this.ENTERED_COMMAND < 0;
148    } // commandCounter()
149
150    /**
151     * go method
152     * @param pDrct the direction
153     * @return 1, 2 or 3 depending on the direction
154     */
155    public int goRoom( final String pDrct )
156    {
157        Room vNextRoom = this.aCurrentRoom.getExit(pDrct);
158        Door vDoor = this.getCurrentRoom().getDoors(pDrct);
159        int vNumber = 2;
160        if ( vNextRoom == null ) 
161            vNumber = 0; 
162        else {
163            if ( vDoor != null && !vDoor.isOpen()){
164                if ( !this.aInventory.containsItem("dragonSoul") ){
165                    return 1;
166                }
167                vDoor.setOpen(true);
168            }
169            this.setNextRoom( vNextRoom );
170            this.printLocationInfo();
171        }
172        return vNumber;
173    } // goRoom(.)
174
175    /**
176    * go method (for random access)
177    * @param pRoom the random room you end up
178    */
179    public void goRandom( final Room pRoom )
180    {
181        this.setNextRoom( pRoom );
182        this.printLocationInfo();
183   } // goRandom(.)
184
185    /**
186     * back method 
187     */
188    public void goBack()
189    {
190        this.setPreviousRooms();
191        this.printLocationInfo();
192    } // goBack(.)
193
194    /**
195     * memorize method 
196     * @return true if the beamer has been used, false otherwise
197     */
198    public boolean memorize()
199    {
200        if ( this.aBeamer.isUsed() )
201            return true;
202        this.aBeamer.charge( this.aCurrentRoom );
203        return false;
204    } // memorize(.)
205
206    /**
207     * teleport the player to the memorized room
208     * @return true if the beamer is charged, false otherwise
209     */
210    public boolean teleport()
211    {
212        if ( this.aBeamer.isCharged() ){
213            this.aCurrentRoom = this.aBeamer.getMemorizedRoom();
214            this.aBeamer.fire();
215            this.printLocationInfo();
216            return true;
217        }
218        return false; 
219    } // teleport(.)
220
221    /**
222     * look method
223     * @param pSW the name of the item
224     * @return true if the item is in the room, false otherwise
225     */
226    public boolean look( final String pSW )
227    {
228        return this.aCurrentRoom.containsItem(pSW);
229    } // look(.)
230
231    /**
232     * eat method
233     * @param pSW the name of the item
234     * @return 0, 1 or 2 depending on the situation 
235     */
236    public int eat( final String pSW )
237    {
238        int vNumber = 2;
239        if ( this.aInventory.containsItem(pSW) ){
240            if ( pSW.equals("stone") ){
241                this.MAX_WEIGHT *= 1e10;
242                vNumber = 0;
243            }
244            else vNumber = 1;
245            this.MAX_WEIGHT -= aInventory.getItem(pSW).getWeight();
246            this.aInventory.removeItem(pSW);
247        }
248        return vNumber;
249    } // eat(.)
250
251    /**
252     * items method
253     * @return the inventory in string format
254     */
255    public String Items()
256    {
257        if ( this.aInventory.hasItem()){
258            StringBuilder vInventory = new StringBuilder( " " );
259            if (this.aInventory.hasItem() ){
260                vInventory.append( "Iventory:" );
261                for ( String vItem : this.aInventory.getIventory().keySet() )
262                    vInventory.append( " " + vItem + " (" + this.aInventory.getItem(vItem).getWeight() + ")" );
263                return vInventory.append( " for a total of "+ this.getWeightCarrying() + "kg" ).toString();
264            }
265            else vInventory.toString();
266        }
267        return "You aren't carrying anything.";
268    } // Items()
269
270    /**
271     * take method
272     * @param pSW the name of the item
273     * @return 0, 1 or 2 depending on the situation 
274     */
275    public int take( final String pSW )
276    {
277            if ( this.aCurrentRoom.containsItem(pSW) ){
278                if ( this.getWeightCarrying() < this.MAX_WEIGHT)
279                    if ( ( this.getWeightCarrying() + this.aCurrentRoom.getItems().getItem(pSW).getWeight() ) > this.MAX_WEIGHT )
280                        return 0;
281                    else {
282                        this.aInventory.setItem( this.getCurrentRoom().getItems().getItem(pSW) );
283                        this.aCurrentRoom.getItems().removeItem(pSW);
284                        return 1;
285                    }
286            }
287            return 2;
288    } // take(.)
289
290    /**
291     * drop method 
292     * @param pSW the name of the item
293     * @return true if the item is in the inventory of the player, false otherwise
294     */
295    public boolean drop( final String pSW )
296    {
297            if ( this.aInventory.containsItem(pSW) ){
298                this.aCurrentRoom.getItems().setItem( this.aInventory.getItem(pSW) );
299                this.aInventory.removeItem(pSW);
300                return true;
301            }
302            return false;
303    } // drop(.)
304} // Player
